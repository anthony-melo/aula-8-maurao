#include <stdio.h>

int somar(int n) {
    if (n <= 1) {
        return 1;
    }
    else {
        return n + somar(n - 1);
    }
}

int main() {
    int numero;

    printf("Digite um numero: ");
    scanf("%d", &numero);

    int resultado = somar(numero);
    printf("A soma de 1 ate %d e: %d\n", numero, resultado);

    return 0;
}
