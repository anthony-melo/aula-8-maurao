#include <stdio.h>

int lanca(int num) {
    if (num < 1) {
        printf("Lançar!\n");
    } else {
        printf("%d...\n", num);
        
        lanca(num - 1);
    }
    return 0;
}

int main() {
    int num;

    printf("Digite um numero para o lancamento: ");
    scanf("%d", &num);

    lanca(num);

    return 0;
}
